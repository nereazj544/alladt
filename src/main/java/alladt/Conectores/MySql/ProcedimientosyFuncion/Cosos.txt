delimiter $$
CREATE FUNCTION nombre_dep (d int)
RETURNS VARCHAR(15)
DETERMINISTIC
BEGIN
  DECLARE nom VARCHAR(15);
  SET nom = 'NO EXISTE';
  SELECT dnombre INTO nom FROM DEPARTAMENTOS WHERE dept_no = d;
  return nom;
END;
